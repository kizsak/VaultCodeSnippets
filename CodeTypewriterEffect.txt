::TypewriterEffect 
{
  (set: $typewriterPos to 1)
  |typewriterOutput>[]

  (live: 20ms)[
    (append: ?typewriterOutput)[(print: $typewriterText's $typewriterPos)]
    (set: $typewriterPos to it + 1)
    (if: $typewriterPos > $typewriterText's length)[
      (stop:)
    ]
  ]
}
